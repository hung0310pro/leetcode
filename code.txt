function reverse($x) {
    if ($x < 0) {
        $a = -1;
        $x = 0 - $x;
    } else {
        $a = 1;
    }
    $n = strlen($x);
    $ret = "";
    for ($i = 0; $i < $n; $i++) {
        $ret .= substr($x, $n - $i - 1, 1);
    }
    $ret = $ret * $a;
    if ($ret >= pow(2, 31) - 1 || $ret <= pow(2, 31) * -1) {
        return 0;
    }
    return $ret;
}
